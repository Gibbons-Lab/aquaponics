Association analysis

Running the analysis

We will start by running an association analysis using DESeq2. In short we will extract our data subsets (particular sample site at a particular time) and get the association with basic plant growth indicators and the inoculum.

library(mbtools)
theme_set(theme_classic())

ps <- readRDS("data/taxonomy.rds")
otu_table(ps)[otu_table(ps) < 1] <- 0
sdata <- fread("data/plant_data_clean.csv")
sdata <- sdata[order(date)][sample == "Growth Bed", .(no_leaves=no_leaves_avg[.N], 
                                                      height=height_avg[.N], 
                                                      root_length=root_length_avg[.N]), 
                                                   by=tank]
old <- fread("data/barcodes.csv")
old[, inoculum := factor(inoculum, levels = c("EIT", "CIT"))]
old[, extraction_kit := factor(extraction_kit)]
sdata <- as.data.frame(sdata[old, on = "tank"])
# Read the greenhouse data
greenhouse <- fread("data/greenhouse.csv")
greenhouse[, date := as.Date(date, format = "%m/%d/%y")]
greenhouse[, time := difftime(date, date[1], units = "days")]
greenhouse[, inoculum := "aquaponics system"]
greenhouse[tank %in% 1:3, inoculum := "CIT"]
greenhouse[, tank := factor(tank)]
greenhouse <- greenhouse[order(-time, tank)]
greenhouse <- greenhouse[, .(nitrate = sum(as.numeric(nitrate)), ammonia = sum(ammonia)), by="tank"]
sdata <- merge(sdata, greenhouse, by = "tank", all = TRUE)
sdata$isb_id[sdata$isb_id == "T0RW"] <- "T0R"
sample_names(ps)[sample_names(ps) == "T0RW"] <- "T0R"
rownames(sdata) <- sdata$isb_id
sample_data(ps) <- sdata
subsets <- list(c("biofilter", "T2"), c("fish feces", "T0"), c("root", "T2"))
vars <- c("no_leaves", "height", "root_length", "inoculum", "nitrate", "ammonia", "extraction_kit", "dead_fish")
tests <- list()
for (s in subsets) {
    psub <- subset_samples(ps, compartment == s[1] & sample_set == s[2])
    res <- association(psub, variables=vars, min_count = 10, in_samples=0.1)
    res[, "site" := s[1]]
    tests[[s[1]]] <- res
}

First let’s have a look which genera associate with our confounders.

confounders <- c("nitrogen", "ammonia", "nitrate", "dead_fish", "extraction_kit")
ggplot(tests[padj < 0.05 & genus %chin% genera & variable %in% confounders],
       aes(x = log2FoldChange, y = reorder(genus, -log2FoldChange),
           color = baseMean)) +
    geom_vline(xintercept = 0, lty = "dashed", color = "black") +
    geom_point(size=2) +
    geom_errorbarh(aes(xmin = log2FoldChange - lfcSE,
                    xmax = log2FoldChange + lfcSE), height = 0.5) +
    facet_grid(site ~ variable, scales = "free_y", space = "free_y") + 
    scale_color_gradient(trans="log10") +

Temporal development

Let’s have a look for the genera that are specifically targeting the root. For that we wil first assemble the corresponding data set:

counts <- taxa_count(ps, zeros = TRUE)
sdata <- as.data.table(sdata)
counts <- counts[sdata, on = c(sample = "isb_id")]
counts[, relative := reads / sum(reads), by = "sample"]

genera <- c("Pseudomonas", "Aeromonas", "Serratia", "Rahnella")
counts <- counts[taxa %in% genera & compartment %in% c("biofilter", "root", "inoculum", "rock wool cubes")]
map <- rep(c("CIT", "EIT"), each = 3)
initial <- counts[is.na(tank)]
initial <- lapply(1:6, function(i) {
  co <- copy(initial) 
  co$tank <- i
  co$inoculum <- map[i]
  co$compartment[co$compartment == "inoculum"] <- "biofilter"
  return(co)
})
initial[[7]] <- counts[!is.na(tank)]
counts <- rbindlist(initial)
counts[, inoculum := factor(inoculum, levels = c("CIT", "EIT"))]
